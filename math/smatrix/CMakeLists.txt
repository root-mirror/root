ROOT_GENERATE_DICTIONARY(G__Smatrix
    Math/SMatrix.h
    Math/SMatrixDfwd.h
    Math/SMatrixFfwd.h
    Math/SVector.h
  MODULE
    Smatrix
  LINKDEF
    LinkDef.h
  OPTIONS
    -writeEmptyRootPCM
  DEPENDENCIES
    Core
    MathCore
)

    Math/SMatrix.h
    Math/SMatrixDfwd.h
    Math/SMatrixFfwd.h
    Math/SVector.h
  MULTIDICT
  MODULE
    Smatrix
  LINKDEF
    LinkDefD32.h

    -writeEmptyRootPCM
  DEPENDENCIES
    Core
    MathCore
)

    G__Smatrix.cxx
    G__Smatrix32.cxx
  DEPENDENCIES
    Core
    MathCore
)
